DESIGN RULES (from Dispatch Board):
  Card:     bg-surface-card border border-surface-border rounded-xl p-5
  Header:   text-text-primary font-semibold text-base + text-text-muted text-sm subtitle
  Row:      flex items-center justify-between px-4 py-3 hover:bg-surface-hover 
            transition-colors cursor-pointer border-b border-surface-border last:border-0
  Name:     text-text-primary font-medium text-sm
  Meta:     text-text-muted text-xs
  Badge:    StatusBadge pattern from CODEX6.9.26.md Section 6
  Add btn:  border border-surface-border text-text-secondary hover:bg-surface-hover
            rounded-lg px-3 py-1.5 text-sm
  Save btn: bg-brand text-text-inverse rounded-lg px-4 py-2 text-sm font-medium
  Input:    bg-surface-base border border-surface-border text-text-primary rounded-lg
            px-3 py-2 placeholder:text-text-muted focus:border-brand
  Section divider: border-t border-surface-border my-6
  No animations on content — transition-colors only
